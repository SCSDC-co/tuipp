52TUIPP_NODISCARD inline TerminalSize
53get_terminal_size()
54{
55 TerminalSize size{};
56
57#ifdef ON_WINDOWS
58 CONSOLE_SCREEN_BUFFER_INFO csbi;
59 GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi);
60 size.width = static_cast<int>(csbi.srWindow.Right - csbi.srWindow.Left + 1);
61 size.height = static_cast<int>(csbi.srWindow.Bottom - csbi.srWindow.Top + 1);
62#elifdef ON_UNIX_LIKE
63 struct winsize w;
64 ioctl(fileno(stdout), TIOCGWINSZ, &w);
65 size.width = static_cast<int>(w.ws_col);
66 size.height = static_cast<int>(w.ws_row);
67#endif
68
69 return size;
70}
71TUIPP_NODISCARD inline bool
72get_support_ansi()
73{
74 char* term = std::getenv("TERM");
75 char* colorterm = std::getenv("COLORTERM");
76
77 if (colorterm) {
78 std::string_view ct(colorterm);
79
80 if (ct == "truecolor" || ct == "24bit") {
81 return true;
82 }
83 }
84
85 if (term) {
86 std::string_view t(term);
87 if (t.find("256color") != std::string_view::npos) {
88 return true;

90 if (t.find("xterm") != std::string_view::npos) {
91 return true;
92 }
93 if (t.find("screen") != std::string_view::npos) {
94 return true;
95 }
96 if (t == "linux") {
97 return true;
98 }
99 }
100
101 return false;

106TUIPP_NODISCARD inline ColorSystem
107get_color_support()
108{
109 if (std::getenv("NO_COLOR")) {
110 return ColorSystem::NO_COLORS;
111 }
112
113#ifdef ON_WINDOWS
114 if (IsWindowsVersionOrGreater(10, 0, 15063)) {
115 return ColorSupport::TRUE_COLOR;
116 }
117#else
118 std::string color_term_env_var = std::getenv("COLORTERM");
119
120 std::transform(
121 color_term_env_var.begin(), color_term_env_var.end(), color_term_env_var.begin(), ::tolower);
122
123 if (!color_term_env_var.empty()) {
124 if (color_term_env_var == "truecolor" || color_term_env_var == "24bit") {
125 return ColorSystem::TRUE_COLOR;
126 }
127 }
128#endif
129
130 return ColorSystem::EIGHT_BIT;
131}
